\subsection{Verluste im Kondensator}
\[ Q_C = \frac{P_B}{P_W} = \tan(\varphi) = \frac{1}{\tan(\delta)} 
\qquad \delta = 90^\circ - \varphi \]
\begin{tabular}{@{}ll}
  $Q_C$: & Güte \\
  $P_B$: & Blindleistung \\
  $P_W$: & Wirkleistung
\end{tabular}
